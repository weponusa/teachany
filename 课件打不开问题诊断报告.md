## 📝 修复脚本

### fix-planned-courses.py

```python
#!/usr/bin/env python3
"""将规划中的课件标记为 gap"""
import json
from pathlib import Path

def fix_planned_courses():
    """将 bio-grade7-* 和 phy-mid-* 课件标记为 gap"""
    
    problem_patterns = [
        ('biology-middle.json', 'bio-grade7-'),
        ('physics-middle.json', 'phy-mid-')
    ]
    
    for tree_file, prefix in problem_patterns:
        tree_path = Path(f'data/trees/{tree_file}')
        
        with open(tree_path, encoding='utf-8') as f:
            tree_data = json.load(f)
        
        def fix_node(node):
            if 'courses' in node and node['courses']:
                # 检查是否包含规划课件
                for course_id in node['courses']:
                    if course_id.startswith(prefix):
                        # 清空 courses 并标记为 gap
                        node['courses'] = []
                        node['status'] = 'gap'
                        print(f"✓ 修复: {course_id} → gap")
            
            # 递归处理子节点
            for key in ['children', 'nodes', 'domains']:
                if key in node:
                    for child in node[key]:
                        fix_node(child)
        
        fix_node(tree_data)
        
        # 保存
        with open(tree_path, 'w', encoding='utf-8') as f:
            json.dump(tree_data, f, ensure_ascii=False, indent=2)
        
        print(f"✅ {tree_file} 修复完成")

if __name__ == '__main__':
    fix_planned_courses()
```

### fix-node-id.py

```python
#!/usr/bin/env python3
"""修复 manifest.json 中的 node_id 不匹配"""
import json
from pathlib import Path

def fix_node_id():
    """将 manifest.node_id 改为与课件 ID 一致"""
    
    for course_dir in Path('examples').iterdir():
        if not course_dir.is_dir():
            continue
        
        manifest_path = course_dir / 'manifest.json'
        if not manifest_path.exists():
            continue
        
        with open(manifest_path, encoding='utf-8') as f:
            manifest = json.load(f)
        
        course_id = course_dir.name
        old_node_id = manifest.get('node_id', '')
        
        # 从课件 ID 推断正确的 node_id
        # 例如：chn-compound-vowel → compound-vowel
        subject_prefix = manifest.get('subject', '')[:3]
        if course_id.startswith(f"{subject_prefix}-"):
            expected_node_id = course_id.replace(f"{subject_prefix}-", '', 1)
        else:
            expected_node_id = course_id
        
        if old_node_id != expected_node_id:
            manifest['node_id'] = course_id  # 直接使用完整的课件 ID
            
            with open(manifest_path, 'w', encoding='utf-8') as f:
                json.dump(manifest, f, ensure_ascii=False, indent=2)
            
            print(f"✓ {course_id}: {old_node_id} → {course_id}")

if __name__ == '__main__':
    fix_node_id()
```
